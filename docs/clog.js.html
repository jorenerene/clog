// https://css-tricks.com/snippets/javascript/get-url-variables/
function getQueryValue (key) {
	var query = window.location.search.substring (1);
	var vars = query.split ("&");
	for (var i = 0; i < vars.length; i++) {
		var pair = vars [i].split ("=");
		
		if (pair [0] == key) {
			return pair [1];
		}
	}

	return false;
}
